from menus.base import Menu, NavigationNode
from menus.menu_pool import menu_pool
from django.utils.translation import ugettext_lazy as _
#

Class for the menu-handling

class tlu_Menu(Menu):
# 

    

  

  

  

    

      

        #
      

      
produce the node-infrastructure for the menu.
Will be rendered in the view 
:param request: The http-request, needed to check for login and user name

    

    

      
    def get_nodes(self, request):

    

  

  

  

    

      

        #
      

      
    

    

      
        nodes = []
        n1 = NavigationNode(_('About me'), "https://tlc-it-consulting.com/index.php/de/ueber-mich", 1)
        n2 = NavigationNode(_('Home'), "/tlu_joyit_game/", 2)
        if request.user.is_authenticated:
            n3 = NavigationNode(request.user.username, "/", 3)
        else:
            n3 = NavigationNode(_("Settings"), "/", 3)
            
        n5 = NavigationNode(_('Change password'), "/accounts/password_change/", 4, 3)
        n6 = NavigationNode("#DIVIDER#","/",5, 3)
        n6_1 = NavigationNode(_('Login'), "/accounts/login/", 6, 3)
        n6_2 = NavigationNode(_('Logout'), "/accounts/logout/", 6, 3)
        nodes.append(n1)
        nodes.append(n2)
        nodes.append(n3)
        nodes.append(n5)
        nodes.append(n6)
        loggedIn=False
        if request is not None:
            if request.user is not None:
                loggedIn=request.user.is_authenticated

        if loggedIn:
            nodes.append(n6_2)
        else:
            nodes.append(n6_1)
            
        return nodes

menu_pool.register_menu(tlu_Menu)
