def log_info(info=""):
# 
    calling_func = sys._getframe(1).f_code.co_name
    thread_name = threading.currentThread().name
    s="<{thread_name}, {calling_func}> {info}".format(thread_name=thread_name,calling_func=calling_func,info=info)
    logging.debug(s)
